#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>
#define MAX_WORDS 500
#define MAX_WORDS_LEN 50

typedef struct{
    char word[MAX_WORDS_LEN];  /////////
    int frequency;
}WordRecord;

int main(){
   char filename[100],ch,tempWord[MAX_WORDS_LEN];   ///////////
   FILE *file;
   int lines=0,words_count=0,characters=0; 
   WordRecord list[MAX_WORDS];
   int uniqueCount=0;

   //1.Read file name from user:
   printf("Enter file name: ");
   scanf("%s",filename);

   //2. Open file
   file=fopen(filename,"r");
   if(file==NULL){
        printf("Error: COuld not open file.\n");
        return 1;
   }

   //3. Read character by character for counts.
   while((ch=fgetc(file))!=EOF){
        characters++;
        if(ch=='\n'){
            lines++;
        }
   }

   //Handle the last line if file doesnt end with \n
   if(characters>0 && ch!='\n'){
        lines++;
   }

   //4. Reset file pointer to beginning for word analysis
   rewind(file);

   //5. Read word by word.
   while(fscanf(file,"%s",tempWord)!=EOF){
        //a. Convert word to lowercase and remove punctuation.
        int j=0;
        char clean[MAX_WORDS_LEN]="";     ////////////////////////////////
        for(int i=0;tempWord[i];i++){
            if(isalpha(tempWord[i])){
                clean[j++]=tolower(tempWord[i]);
            }
        }
        clean[j]='\0';
        if(strlen(clean)==0){
            continue;
        }
        //d. Increment total word count.
        words_count++;
        //b & c. Store word and track frequency.
        int found=0;
        for(int i=0;i<uniqueCount;i++){
            if(strcpy(list[i].word,clean)==0){
                list[uniqueCount].frequency++;
                found=1;
                break;
            }
        }
        if(!found && uniqueCount<MAX_WORDS){
            strcpy(list[uniqueCount].word,clean);
            list[uniqueCount].frequency=1;
            uniqueCount++;
        }
   }

   //6. Close the file.
    fclose(file);

    //7. Find word with highest frequency.
    int maxFreq=0;
    char mostFrequent[MAX_WORDS_LEN]="";
    for(int i=0;i<uniqueCount;i++){
        if(list[i].frequency>maxFreq){
            maxFreq=list[i].frequency;
            strcpy(mostFrequent,list[i].word);
        }
    }

    //8. Display results

    printf("\n---Manuscripts Analysis---\n");
    printf("Total characters: %d\n",characters);
    printf("Total words: %d\n",words_count);
    printf("Total lines: %d\n",lines);
    if(uniqueCount>0){
        printf("Most Frequent: '%s'(found %d times)\n",mostFrequent,maxFreq);
    }
    return 0;
}
